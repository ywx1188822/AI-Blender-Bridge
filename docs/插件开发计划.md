```python
# SDNode/plugins/ai_blender_bridge/short_drama.py
import bpy
import requests
import json

class AIBRIDGE_OT_GenerateScript(bpy.types.Operator):
    """生成短剧剧本"""
    
    bl_idname = "aibridge.generate_script"
    bl_label = "生成剧本 📝"
    bl_description = "使用 LLM 生成短剧剧本"
    
    theme: bpy.props.StringProperty(name="主题", default="AI 觉醒之日")
    duration: bpy.props.IntProperty(name="时长 (秒)", default=180)
    style: bpy.props.EnumProperty(
        name="风格",
        items=[
            ("sci-fi", "科幻", "Science Fiction"),
            ("drama", "剧情", "Drama"),
            ("comedy", "喜剧", "Comedy"),
            ("action", "动作", "Action")
        ]
    )
    
    def execute(self, context):
        # 调用 LLM API 生成剧本
        script = self.call_llm_api(self.theme, self.duration, self.style)
        
        # 保存剧本
        self.save_script(script)
        
        self.report({'INFO'}, f"✅ 剧本生成完成：{self.theme}")
        return {'FINISHED'}
    
    def call_llm_api(self, theme, duration, style):
        """调用 LLM API 生成剧本"""
        from .llm_api import LLMGateway
        
        prompt = f"""
        请创作一个{duration}秒的短剧剧本。
        
        主题：{theme}
        风格：{style}
        
        要求：
        1. 包含 3-5 个场景
        2. 每个场景有详细的镜头描述
        3. 包含角色对话
        4. 输出 JSON 格式
        """
        
        llm = LLMGateway()
        response = llm.generate(prompt)
        
        return json.loads(response)
    
    def save_script(self, script):
        """保存剧本到文件"""
        import json
        from pathlib import Path
        
        output_dir = Path(bpy.path.abspath("//")) / "scripts"
        output_dir.mkdir(exist_ok=True)
        
        script_file = output_dir / f"{self.theme}.json"
        with open(script_file, 'w', encoding='utf-8') as f:
            json.dump(script, f, ensure_ascii=False, indent=2)
        
        print(f"📄 剧本已保存：{script_file}")

def register():
    bpy.utils.register_class(AIBRIDGE_OT_GenerateScript)

def unregister():
    bpy.utils.unregister_class(AIBRIDGE_OT_GenerateScript)
```

### 4. LLM API 网关 (复用现有代码)

```python
# SDNode/plugins/ai_blender_bridge/llm_api.py
import requests
import json
from pathlib import Path

class LLMGateway:
    """LLM API 统一网关"""
    
    def __init__(self, provider="aliyun"):
        self.provider = provider
        self.config = self.load_config()
    
    def load_config(self):
        """加载 API 配置"""
        config_file = Path(__file__).parent / "config" / "api_config.json"
        if config_file.exists():
            with open(config_file, 'r', encoding='utf-8') as f:
                return json.load(f)
        return {}
    
    def generate(self, prompt, **kwargs):
        """生成文本"""
        if self.provider == "aliyun":
            return self.call_aliyun(prompt, **kwargs)
        elif self.provider == "kimi":
            return self.call_kimi(prompt, **kwargs)
        elif self.provider == "glm":
            return self.call_glm(prompt, **kwargs)
        else:
            raise ValueError(f"不支持的提供商：{self.provider}")
    
    def call_aliyun(self, prompt, **kwargs):
        """调用阿里云通义千问"""
        url = "https://dashscope.aliyuncs.com/api/v1/services/aigc/text-generation/generation"
        headers = {
            "Authorization": f"Bearer {self.config['aliyun']['api_key']}",
            "Content-Type": "application/json"
        }
        payload = {
            "model": "qwen-plus",
            "input": {"messages": [{"role": "user", "content": prompt}]},
            "parameters": kwargs
        }
        
        response = requests.post(url, headers=headers, json=payload)
        response.raise_for_status()
        result = response.json()
        
        return result["output"]["text"]
    
    def call_kimi(self, prompt, **kwargs):
        """调用 Kimi API"""
        url = "https://api.moonshot.cn/v1/chat/completions"
        headers = {
            "Authorization": f"Bearer {self.config['kimi']['api_key']}",
            "Content-Type": "application/json"
        }
        payload = {
            "model": "moonshot-v1-8k",
            "messages": [{"role": "user", "content": prompt}],
            **kwargs
        }
        
        response = requests.post(url, headers=headers, json=payload)
        response.raise_for_status()
        result = response.json()
        
        return result["choices"][0]["message"]["content"]
```

### 5. ComfyUI 工作流集成

```python
# SDNode/plugins/ai_blender_bridge/character_gen.py
import bpy
import requests
import json

class AIBRIDGE_OT_GenerateCharacter(bpy.types.Operator):
    """生成角色"""
    
    bl_idname = "aibridge.generate_character"
    bl_label = "生成角色 👤"
    bl_description = "使用 ComfyUI 生成角色"
    
    prompt: bpy.props.StringProperty(name="提示词", default="1girl, silver hair, blue eyes, school uniform")
    negative_prompt: bpy.props.StringProperty(name="负向提示词", default="ugly, deformed, noisy, blurry")
    steps: bpy.props.IntProperty(name="步数", default=20)
    cfg_scale: bpy.props.FloatProperty(name="CFG Scale", default=7.0)
    seed: bpy.props.IntProperty(name="种子", default=-1)
    
    def execute(self, context):
        # 加载角色生成工作流
        workflow = self.load_character_workflow()
        
        # 修改提示词参数
        workflow["3"]["inputs"]["text"] = self.prompt
        workflow["4"]["inputs"]["text"] = self.negative_prompt
        workflow["5"]["inputs"]["seed"] = self.seed if self.seed != -1 else self.random_seed()
        workflow["5"]["inputs"]["steps"] = self.steps
        workflow["5"]["inputs"]["cfg"] = self.cfg_scale
        
        # 提交到 ComfyUI
        prompt_id = self.submit_to_comfyui(workflow)
        
        # 等待完成
        images = self.wait_for_completion(prompt_id)
        
        # 加载到 Blender
        self.load_images_to_blender(images)
        
        self.report({'INFO'}, "✅ 角色生成完成")
        return {'FINISHED'}
    
    def load_character_workflow(self):
        """加载角色生成工作流"""
        from pathlib import Path
        workflow_file = Path(__file__).parent / "comfyui_workflows" / "character_card.json"
        
        with open(workflow_file, 'r', encoding='utf-8') as f:
            return json.load(f)
    
    def submit_to_comfyui(self, workflow):
        """提交到 ComfyUI"""
        server = bpy.context.scene.aibridge_settings.comfyui_server
        url = f"http://{server}/prompt"
        
        import uuid
        client_id = str(uuid.uuid4())
        
        response = requests.post(url, json={
            "prompt": workflow,
            "client_id": client_id
        })
        response.raise_for_status()
        result = response.json()
        
        return result["prompt_id"]
    
    def wait_for_completion(self, prompt_id, timeout=300):
        """等待生成完成"""
        import time
        
        server = bpy.context.scene.aibridge_settings.comfyui_server
        url = f"http://{server}/history/{prompt_id}"
        
        start_time = time.time()
        while time.time() - start_time < timeout:
            response = requests.get(url)
            response.raise_for_status()
            history = response.json()
            
            if prompt_id in history:
                outputs = history[prompt_id].get("outputs", {})
                images = []
                
                for node_id, node_out in outputs.items():
                    if "images" in node_out:
                        for img in node_out["images"]:
                            if "filename" in img and "subfolder" in img:
                                images.append({
                                    "filename": img["filename"],
                                    "subfolder": img["subfolder"],
                                    "type": img.get("type", "output")
                                })
                
                if images:
                    return images
            
            time.sleep(2)
        
        raise TimeoutError("生成超时")
    
    def load_images_to_blender(self, images):
        """加载图片到 Blender"""
        server = bpy.context.scene.aibridge_settings.comfyui_server
        
        for img in images:
            url = f"http://{server}/view"
            params = {
                "filename": img["filename"],
                "subfolder": img["subfolder"],
                "type": img["type"]
            }
            
            response = requests.get(url, params=params)
            response.raise_for_status()
            
            # 保存图片到本地
            from pathlib import Path
            output_dir = Path(bpy.path.abspath("//")) / "output" / "characters"
            output_dir.mkdir(parents=True, exist_ok=True)
            
            img_file = output_dir / img["filename"]
            with open(img_file, 'wb') as f:
                f.write(response.content)
            
            # 加载为 Blender 图像
            bpy.ops.image.open(filepath=str(img_file))
        
        print(f"🖼️ 已加载 {len(images)} 张图片到 Blender")
    
    def random_seed(self):
        """生成随机种子"""
        import random
        return random.randint(0, 2**32 - 1)

def register():
    bpy.utils.register_class(AIBRIDGE_OT_GenerateCharacter)

def unregister():
    bpy.utils.unregister_class(AIBRIDGE_OT_GenerateCharacter)
```
