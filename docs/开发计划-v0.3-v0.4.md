```python
    @staticmethod
    def handle_api_error(error, context=""):
        """处理 API 错误"""
        error_map = {
            401: "API Key 无效或过期",
            403: "权限不足",
            404: "资源不存在",
            429: "请求过于频繁，请稍后重试",
            500: "服务器内部错误",
            503: "服务暂时不可用"
        }
        return error_map.get(error.status_code, f"未知错误：{error}")
    
    @staticmethod
    def handle_timeout_error(timeout, operation=""):
        """处理超时错误"""
        return f"{operation}超时 ({timeout}秒)，请检查网络或服务器状态"
```

```python
import sqlite3
from pathlib import Path

class AIBRIDGE_Database:
    """历史记录数据库"""
    
    def __init__(self):
        db_path = Path(__file__).parent / "aibridge_history.db"
        self.conn = sqlite3.connect(db_path)
        self.create_tables()
    
    def create_tables(self):
        """创建数据表"""
        cursor = self.conn.cursor()
        
        # 生成历史表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS generations (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                type TEXT NOT NULL,  -- character/video/script
                prompt TEXT,
                negative_prompt TEXT,
                parameters TEXT,  -- JSON
                output_files TEXT,  -- JSON
                status TEXT,  -- pending/running/completed/failed
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                completed_at TIMESTAMP
            )
        """)
        
        # 任务队列表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS tasks (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                prompt_id TEXT,
                type TEXT,
                status TEXT,
                progress REAL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        self.conn.commit()
```

```python
import subprocess

class AIBRIDGE_FFmpeg:
    """FFmpeg 工具类"""
    
    @staticmethod
    def stitch_videos(video_files, output, transition="fade", duration=0.5):
        """拼接视频"""
        # 创建 ffmpeg 命令
        cmd = [
            "ffmpeg",
            "-i", "concat:" + "|".join(video_files),
            "-c", "copy",
            output
        ]
        subprocess.run(cmd, check=True)
    
    @staticmethod
    def add_music(video, music, output, volume=0.3):
        """添加背景音乐"""
        cmd = [
            "ffmpeg",
            "-i", video,
            "-i", music,
            "-c:v", "copy",
            "-c:a", "aac",
            "-filter:a", f"volume={volume}",
            output
        ]
        subprocess.run(cmd, check=True)
```

#### 4.1 异步任务处理

```python
# utils/async_tasks.py (新增模块)

import threading
import queue

class AIBRIDGE_TaskQueue:
    """异步任务队列"""
    
    def __init__(self, max_workers=4):
        self.queue = queue.Queue()
        self.workers = []
        self.max_workers = max_workers
        self.start_workers()
    
    def start_workers(self):
        """启动工作线程"""
        for _ in range(self.max_workers):
            worker = threading.Thread(target=self._worker_loop, daemon=True)
            worker.start()
            self.workers.append(worker)
    
    def submit(self, task, callback=None):
        """提交任务"""
        self.queue.put((task, callback))
    
    def _worker_loop(self):
        """工作线程循环"""
        while True:
            task, callback = self.queue.get()
            try:
                result = task()
                if callback:
                    callback(result, None)
            except Exception as e:
                if callback:
                    callback(None, e)
            self.queue.task_done()
```

```python
# utils/cache.py (新增模块)

class AIBRIDGE_Cache:
    """缓存系统"""
    
    def __init__(self, max_size=100):
        self.cache = {}
        self.max_size = max_size
    
    def get(self, key):
        """获取缓存"""
        return self.cache.get(key)
    
    def set(self, key, value, ttl=3600):
        """设置缓存"""
        import time
        self.cache[key] = {
            "value": value,
            "expires": time.time() + ttl
        }
        self._cleanup()
    
    def _cleanup(self):
        """清理过期缓存"""
        import time
        now = time.time()
        expired = [k for k, v in self.cache.items() if v["expires"] < now]
        for k in expired:
            del self.cache[k]
```
